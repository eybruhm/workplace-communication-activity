BEGIN
    DISPLAY "Enter No. of Days:"
    INPUT noOfDays
    DISPLAY "Enter Rate per Day:"
    INPUT ratePerDay
    DISPLAY "Enter Overtime Hours:"
    INPUT otHours
    DISPLAY "Enter Rate per OT Hour:"
    INPUT otRate
    DISPLAY "Enter Add Allowance:"
    INPUT addAllowance

    regIncome = noOfDays * ratePerDay
    otIncome = otHours * otRate
    grIncome = regIncome + otIncome + addAllowance

    DISPLAY "Regular Income: ", regIncome
    DISPLAY "Overtime Income: ", otIncome
    DISPLAY "Gross Income: ", grIncome

    DISPLAY "Enter SSS/GSIS Contribution:"
    INPUT sss
    DISPLAY "Enter Pagibig Contribution:"
    INPUT pagibig
    DISPLAY "Enter Philhealth Contribution:"
    INPUT philhealth
    DISPLAY "Enter Cash Advance:"
    INPUT cashAdvance
    DISPLAY "Enter Cash Bond:"
    INPUT cashBond
    DISPLAY "Enter Others:"
    INPUT others

    tax = grIncome * 0.10
    tDeductions = sss + pagibig + philhealth + cashAdvance + cashBond + others + tax
    netPay = grIncome - tDeductions

    DISPLAY "W/Holding Tax: ", tax
    DISPLAY "Total Deductions: ", tDeductions
    DISPLAY "Net Pay: ", netPay
END
